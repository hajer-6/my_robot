243  def build_write_command(self, address, data):
244 
245  try:

275  def check_response(self, response):
276 
277  state = RESPONSE_ERROR
278 
279  if len(response) > 1:
280 
281  response_header = response[0]
282  response_status = response[1]
283 
284  if response_header == RESPONSE_HEADER:
285 
286  if response_status == WRITE_SUCCESS:
287  self._print("Write succes")
288  state = RESPONSE_OK
289 
290  if response_status == READ_FAIL:
291  self._print("Read fail")
292  state = RESPONSE_ERROR
293 
294  if response_status == WRITE_FAIL:
295  self._print("Write fail")
296  state = RESPONSE_ERROR
297 
298  if response_status == REGMAP_INVALID_ADDRESS:
299  self._print("Regmap invalid address")
300  state = RESPONSE_ERROR
301 
302  if response_status == REGMAP_WRITE_DISABLED:
303  self._print("Regmap write disabled")
304  state = RESPONSE_ERROR
305 
306  if response_status == WRONG_START_BYTE:
307  self._print("Wrong start byte")
308  state = RESPONSE_ERROR
309 
310  if response_status == BUS_OVER_RUN_ERROR:
311  self._print("Bus over run error")
312  state = RESPONSE_ERROR
313 
314  if response_status == MAX_LENGTH_ERROR:
315  self._print("Max length error")
316  state = RESPONSE_ERROR
317 
318  if response_status == MIN_LENGTH_ERROR:
319  self._print("Min length error")
320  state = RESPONSE_ERROR
321 
322  if response_status == RECEIVE_CHARACTER_TIMEOUT:
323  self._print("Receive character timeout")
324  state = RESPONSE_ERROR
325 
326  elif response_header == READ_SUCCESS:
327  self._print("Read success")
328  state = RESPONSE_OK
329 
330  else:
331  self._print("Response header not detected")
332  state = RESPONSE_ERROR
333 
334  else:

336  state = RESPONSE_ERROR

341  def write_imu(self, address, data):
342 
343  status = -1
344  attempts = 0
345 
346  while status != RESPONSE_OK:
347 
348  # Write register with data value
349  command = self.build_write_command(address, data)
350  self.serial_port.write(command)
351 
352  # Get serial response
353  command_length = 2 # Response is always made up of two bytes
354  response = self.serial_port.read(command_length)
355 
356  # Check response
357  status = self.check_response(response)
358 
359  attempts+=1
360  if attempts >= 10:
361  self._print("Error, after ten attempts the response was not received correctly")
362  status = RESPONSE_ERROR
363  break
364 
365 
366 
367  return response, status

452  def reset_imu(self):
453 
454  # Reset does not return a standard result, then write_imu
455  # and read_imu are not used
456 
457  # Reset IMU
458  reset_command = self.build_write_command(SYS_TRIGGER, RST_SYS)
459  self.serial_port.write(reset_command)
460 
461  # Prepare the command to query the ID
462  id_command = self.build_read_command(CHIP_ID, 1)
463  response = []
464 
465  # Wait until the IMU is ready again. If the IMU is still
466  # resetting it only returns 1 byte for any command
467  while len(response) < 2:
468  self.serial_port.write(id_command)
469  response = self.serial_port.read(3)
470  time.sleep(0.1)
471 
472  self._print("IMU successfully reset")
473 
474  # If the reset fails, the IMU stays in the while loop
475  status = RESPONSE_OK
476 
477  return status
